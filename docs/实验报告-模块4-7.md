```csharp
public ServiceResult<int> SaveFamilyInfo(FamilyInfo familyInfo)
{
    var validation = ValidateLinkedStudent(familyInfo.StudentNo);
    if (!validation.Succeeded)
        return ServiceResult<int>.Failure(validation.Message);

    if (!InputValidator.ValidateName(familyInfo.RelationName) || string.IsNullOrWhiteSpace(familyInfo.Relationship))
        return ServiceResult<int>.Failure("请填写关系人姓名和关系");

    if (!string.IsNullOrWhiteSpace(familyInfo.PhoneNumber) && !InputValidator.ValidatePhone(familyInfo.PhoneNumber))
        return ServiceResult<int>.Failure("联系电话格式错误");

    familyInfo.StudentId = validation.Data;
    var id = repository.SaveFamilyInfo(familyInfo);
    AppLogger.Info($"保存家庭信息：{familyInfo.StudentNo} {familyInfo.RelationName}");
    return ServiceResult<int>.Success(id, "家庭信息保存成功");
}
```

**文件：`Data/Repositories/StudentRepository.cs`（搜索学生）**

```csharp
public List<Student> Search(string keyword)
{
    keyword = keyword.Trim();
    if (string.IsNullOrWhiteSpace(keyword))
        return GetAll();
    using var context = StudentsManagerDbContextFactory.CreateDbContext();
    return context.Students.AsNoTracking()
        .Where(student =>
            EF.Functions.Like(student.StudentNo, $"%{keyword}%") ||
            EF.Functions.Like(student.Name, $"%{keyword}%") ||
            EF.Functions.Like(student.Department, $"%{keyword}%") ||
            EF.Functions.Like(student.Major, $"%{keyword}%") ||
            EF.Functions.Like(student.Class, $"%{keyword}%"))
        .OrderBy(student => student.StudentNo)
        .ToList();
}
```

**学号格式错误导致保存失败**

新增学生时输入不合法格式的学号，系统拒绝保存。

#### 问题1（错误注入方式）

**参数错误注入**

在新增学生窗口中输入不符合规则的学号：
- 学号：`12`（长度不足 4 位）
- 或学号：`@#$%`（包含非法字符）

#### 问题1（复现步骤）

1. 以 `admin` 登录
2. 点击"学生档案管理" → "新增"
3. 学号输入 `12`（仅 2 位数字）
4. 姓名输入 `测试`
5. 填写其他必填字段
6. 点击"保存"

#### 问题1（截图）

**说明**：截图保存时弹出的错误提示，显示"请检查学号和姓名格式"或类似校验提示信息。

#### 问题1（原因分析）

**分析**：`InputValidator.ValidateStudentNo()` 方法：

```csharp
public static bool ValidateStudentNo(string studentNo)
{
    return !string.IsNullOrWhiteSpace(studentNo) && 
           Regex.IsMatch(studentNo, @"^[A-Za-z0-9]{4,20}$");
}
```

正则要求学号为 4-20 位字母或数字。输入 `12` 只有 2 位，不匹配正则，校验失败。

**涉及模块**：模块四（学生档案管理）+ 通用校验模块

**涉及代码**：
- `Common/InputValidator.cs` 第 6-10 行（`ValidateStudentNo`）
- `Services/StudentService.cs` 中的 Validata 方法

#### 问题1（解决方案）

**方案**：输入符合规则的学号，例如 `2024013`（4-20 位字母或数字）。

#### 问题1（解决方案-正确截图）

#### 问题3（错误注入方式）

**代码错误注入**

在 `StudentRepository` 的删除方法中注释掉级联删除代码：

```csharp
public void Delete(int id)
{
    using var context = StudentsManagerDbContextFactory.CreateDbContext();
    var student = context.Students.Find(id);
    if (student == null) return;

    // 注释掉级联删除
    // var familyInfos = context.FamilyInfos.Where(f => f.StudentId == id).ToList();
    // context.FamilyInfos.RemoveRange(familyInfos);
    // ... 其他级联删除

    context.Students.Remove(student);
    context.SaveChanges();
}
```

#### 问题3（复现步骤）

1. 在 `StudentRepository` 中注释级联删除代码
2. 以 `admin` 登录
3. 点击"学生档案管理"
4. 选中一个学生（如 2024001 张三）
5. 点击"删除"
6. 在确认框中点击"是"

#### 问题3（截图）

**说明**：截图删除后的学生列表，张三已不在列表中。但打开"家庭信息" Tab，仍能看到张三的家庭联系人记录。

#### 问题3（原因分析）

**分析**：`StudentRepository` 的删除方法原本包含了对关联表的清理：

```csharp
// 正常时应包含
context.FamilyInfos.RemoveRange(context.FamilyInfos.Where(f => f.StudentId == id));
context.RewardRecords.RemoveRange(context.RewardRecords.Where(r => r.StudentId == id));
// ... 等
```

注释掉后，只删除了 `Students` 表中的主记录，但 `FamilyInfos`、`RewardRecords`、`PunishmentRecords`、`HealthRecords`、`StudentRegistrations`、`StatusChangeRecords`、`ScholarshipInfos`、`GraduationInfos`、`Scores` 中的关联记录未被删除，成为孤儿数据。

**涉及模块**：模块四（学生档案管理）

**涉及代码**：`Data/Repositories/StudentRepository.cs` 中的 `Delete()` 方法

#### 问题3（解决方案）

**方案**：恢复级联删除代码，确保删除学生时同时清理所有关联表数据。

```csharp
public void Delete(int id)
{
    using var context = StudentsManagerDbContextFactory.CreateDbContext();
    var student = context.Students.Find(id);
    if (student == null) return;

    context.FamilyInfos.RemoveRange(context.FamilyInfos.Where(f => f.StudentId == id));
    context.RewardRecords.RemoveRange(context.RewardRecords.Where(r => r.StudentId == id));
    context.PunishmentRecords.RemoveRange(context.PunishmentRecords.Where(p => p.StudentId == id));
    context.HealthRecords.RemoveRange(context.HealthRecords.Where(h => h.StudentId == id));
    context.StudentRegistrations.RemoveRange(context.StudentRegistrations.Where(r => r.StudentId == id));
    context.StatusChangeRecords.RemoveRange(context.StatusChangeRecords.Where(c => c.StudentId == id));
    context.ScholarshipInfos.RemoveRange(context.ScholarshipInfos.Where(s => s.StudentId == id));
    context.GraduationInfos.RemoveRange(context.GraduationInfos.Where(g => g.StudentId == id));
    context.Scores.RemoveRange(context.Scores.Where(s => s.StudentId == id));

    context.Students.Remove(student);
    context.SaveChanges();
}
```

#### 问题3（解决方案-正确截图）

**说明**：修复后删除张三，再到"家庭信息" Tab 中查看，张三的家庭信息已不存在。

```csharp
public ServiceResult<int> SaveRegistration(StudentRegistration record)
{
    if (record.Id != 0 && !repository.RegistrationExists(record.Id))
        return ServiceResult<int>.Failure("当前记录已不存在，请刷新后重试");
    if (repository.RegistrationDuplicateExists(record))
        return ServiceResult<int>.Failure("该学号已经存在登记记录");
    if (record.Id == 0)
    {
        record.Id = repository.AddRegistration(record);
        AppLogger.Info($"新增学籍登记：{record.StudentNo}");
    }
    else
    {
        repository.UpdateRegistration(record);
        AppLogger.Info($"更新学籍登记：{record.StudentNo}");
    }
    return ServiceResult<int>.Success(record.Id);
}
```

```csharp
public bool RegistrationDuplicateExists(StudentRegistration record)
{

        item.Semester == record.Semester &&
```

```csharp
private static ServiceResult Validate(Score score)
{
    if (!InputValidator.ValidateStudentNo(score.StudentNo) || !InputValidator.ValidateName(score.StudentName))
        return ServiceResult.Failure("请检查学号和姓名格式");
    if (!InputValidator.ValidateAcademicYear(score.AcademicYear) || !InputValidator.ValidateSemester(score.Semester))
        return ServiceResult.Failure("请检查学年和学期格式");
    if (string.IsNullOrWhiteSpace(score.CourseNo) || string.IsNullOrWhiteSpace(score.CourseName) || score.Credit < 0)
        return ServiceResult.Failure("请检查课程和学分信息");
    if (!IsScoreInRange(score.RegularScore) || !IsScoreInRange(score.ExamScore) || !IsScoreInRange(score.TotalScore))
        return ServiceResult.Failure("成绩应在0-100之间");
    return ServiceResult.Success();
}

private static bool IsScoreInRange(decimal? score) => !score.HasValue || score is >= 0 and <= 100;
```

```csharp
public List<Score> GetByAcademicYearSemester(string academicYear, string semester)

        .Where(score => score.AcademicYear == academicYear && score.Semester == semester)

        .ToList();
```

```csharp
private void ApplyStudentFilter()
{
    var filtered = students.AsEnumerable();
    var queryType = (cmbQueryType.SelectedItem as string) ?? string.Empty;
    var keyword = txtKeyword.Text.Trim();

    if (!string.IsNullOrWhiteSpace(keyword))
    {
        switch (queryType)
        {
            case "学号":
                filtered = filtered.Where(s => s.StudentNo.Contains(keyword));
                break;
            case "姓名":
                filtered = filtered.Where(s => s.Name.Contains(keyword));
                break;
            case "院系":
                filtered = filtered.Where(s => s.Department.Contains(keyword));
                break;
            case "班级":
                filtered = filtered.Where(s => s.Class.Contains(keyword));
                break;
        }
    }

    var resultList = filtered.ToList();
    dataGrid.ItemsSource = resultList;
    // 计算统计摘要
    txtResultCount.Text = $"当前结果总数：{resultList.Count}";
    txtMaleCount.Text = $"男生数量：{resultList.Count(s => s.Gender == "男")}";
    txtFemaleCount.Text = $"女生数量：{resultList.Count(s => s.Gender == "女")}";
    txtTopDepartment.Text = $"人数最多的院系：{resultList.GroupBy(s => s.Department).OrderByDescending(g => g.Count()).FirstOrDefault()?.Key ?? "无"}";
}
```
